new Promise<QueryResults>((resolve, reject) => {
    client.eachRow(
      query,
      params,
      queryOptions,
      (n, result) => {
        resultsArray.push(result as unknown as UserSettingsDBRow);
      },
      (err, result) => {
        if (err) {
          reject(err);
        } else {
          queryResults.settings = resultsArray;
          queryResults.pageState = result.pageState;
          resolve(queryResults);
        }
      },
    );
  })
